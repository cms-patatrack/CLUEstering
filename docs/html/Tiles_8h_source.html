87template <uint8_t Ndim>
88int tiles<Ndim>::getBin(float coord_, int dim_) const {
89 int coord_Bin{static_cast<int>((coord_ - minMax[dim_][0]) / tilesSize[dim_])};
90 coord_Bin = std::min(coord_Bin, static_cast<int>(std::pow(nTiles, 1.0 / Ndim) - 1));
91 coord_Bin = std::max(coord_Bin, 0);
92 return coord_Bin;
93}

95template <uint8_t Ndim>
96int tiles<Ndim>::getGlobalBin(std::vector<float> const& coords) const {
97 int globalBin{getBin(coords[0], 0)};
98 int nTilesPerDim{static_cast<int>(std::pow(nTiles, 1.0 / Ndim))};
99 for (int i{1}; i != Ndim; ++i) {
100 globalBin += nTilesPerDim * getBin(coords[i], i);
101 }
102 return globalBin;

134void tiles<Ndim>::searchBox(std::array<std::vector<int>, Ndim> const& xjBins,
135 std::array<int, Ndim>& comb,
136 std::vector<int>& sBox) {
137 if constexpr (N_ == Ndim) {
138 sBox.push_back(getGlobalBinByBin(comb));
139 } else {
140 for (int x : xjBins[N_]) {
141 comb[N_] = x;

144 }

156std::vector<int> tiles<Ndim>::getBinsFromRange(float x_min, float x_max, int dim_) {
157 int minBin{getBin(x_min, dim_)};
158 int maxBin{getBin(x_max, dim_)};
159
160 std::vector<int> listOfBins(maxBin - minBin + 1);
161 std::iota(listOfBins.begin(), listOfBins.end(), minBin);
162 return listOfBins;
