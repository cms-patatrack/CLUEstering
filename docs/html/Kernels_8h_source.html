40class gaussianKernel : public kernel {
41private:
42 float m_gaus_avg;
43 float m_gaus_std;
44 float m_gaus_amplitude;
45
46public:
47 gaussianKernel(float gaus_avg, float gaus_std, float gaus_amplitude)
48 : m_gaus_avg{gaus_avg}, m_gaus_std{gaus_std}, m_gaus_amplitude{gaus_amplitude} {}
49
50 float operator()(float dist_ij, int point_id, int j) const override {
51 if (point_id == j) {
52 return 1.f;
53 } else {
54 return static_cast<float>(
55 m_gaus_amplitude * std::exp(-std::pow(dist_ij - m_gaus_avg, 2) / (2 * std::pow(m_gaus_std, 2))));
56 }
57 }
58};

60class exponentialKernel : public kernel {

70 if (point_id == j) {
71 return 1.f;
72 } else {
