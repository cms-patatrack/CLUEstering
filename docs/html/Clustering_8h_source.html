168template <uint8_t Ndim>
169std::array<float, Ndim> ClusteringAlgo<Ndim>::calculateTileSize(int NTiles, tiles<Ndim>& tiles_) {
170 std::array<float, Ndim> tileSizes;
171 int NperDim{static_cast<int>(std::pow(NTiles, 1.0 / Ndim))};
172
173 for (int i{}; i != Ndim; ++i) {
174 float tileSize;
175 float dimMax{*std::max_element(points_.coordinates_[i].begin(), points_.coordinates_[i].end())};
176 float dimMin{*std::min_element(points_.coordinates_[i].begin(), points_.coordinates_[i].end())};
177 tiles_.minMax[i] = {dimMin, dimMax};
178 tileSize = (dimMax - dimMin) / NperDim;
179
180 tileSizes[i] = tileSize;
181 }
182 return tileSizes;

213template <uint8_t Ndim>
214void ClusteringAlgo<Ndim>::calculateLocalDensity(tiles<Ndim>& tiles, kernel const& ker) {
215 // loop over all points
216 for (int i{}; i < points_.n; ++i) {
217 // get search box
218 std::array<std::vector<int>, Ndim> xjBins;
219 for (int j{}; j != Ndim; ++j) {
220 xjBins[j] =
221 tiles.getBinsFromRange(points_.coordinates_[j][i] - dc_, points_.coordinates_[j][i] + dc_, j);
222
223 // Overflow
224 if (points_.coordinates_[j][i] + dc_ > domains_[j].max) {
225 std::vector<int> overflowBins =
226 std::move(tiles.getBinsFromRange(domains_[j].min, domains_[j].min + dc_, j));
227 xjBins[j].insert(xjBins[j].end(), overflowBins.begin(), overflowBins.end());
228 // Underflow
229 } else if (points_.coordinates_[j][i] - dc_ < domains_[j].min) {
230 std::vector<int> underflowBins =
231 std::move(tiles.getBinsFromRange(domains_[j].max - dc_, domains_[j].max, j));
232 xjBins[j].insert(xjBins[j].end(), underflowBins.begin(), underflowBins.end());
233 }
234 }
235 std::vector<int> search_box;
236 tiles.searchBox(xjBins, search_box);
237
238 // loop over bins in the search box
239 for (int binId : search_box) {
240 // iterate over the points inside this bin
241 for (auto binIterId : tiles[binId]) {
242 int j{binIterId};
243 // query N_{dc_}(i)
244 float dist_ij{distance(i, j)};
245
246 if (dist_ij <= dc_) {
247 // sum weights within N_{dc_}(i) using the chosen kernel
248 points_.rho[i] += ker(dist_ij, i, j) * points_.weight[j];
249 }
250 } // end of interate inside this bin
251 } // end of loop over bins in the search box
252 } // end of loop over points
253}

255template <uint8_t Ndim>
256void ClusteringAlgo<Ndim>::calculateDistanceToHigher(tiles<Ndim>& tiles) {
257 float dm{outlierDeltaFactor_ * dc_};
258
259 // loop over all points
260 for (int i{}; i < points_.n; ++i) {
261 // default values of delta and nearest higher for i
262 float delta_i{std::numeric_limits<float>::max()};
263 int nearestHigher_i{-1}; // if this doesn't change, the point is either a seed or an outlier
264 float rho_i{points_.rho[i]};
265
266 // get search box
267 std::array<std::vector<int>, Ndim> xjBins;
268 for (int j{}; j != Ndim; ++j) {
269 xjBins[j] =
270 tiles.getBinsFromRange(points_.coordinates_[j][i] - dm, points_.coordinates_[j][i] + dm, j);
271
272 // Overflow
273 if (points_.coordinates_[j][i] + dm > domains_[j].max) {
274 std::vector<int> overflowBins =
275 std::move(tiles.getBinsFromRange(domains_[j].min, domains_[j].min + dm, j));
276 xjBins[j].insert(xjBins[j].end(), overflowBins.begin(), overflowBins.end());
277 // Underflow
278 } else if (points_.coordinates_[j][i] - dm < domains_[j].min) {
279 std::vector<int> underflowBins =
280 std::move(tiles.getBinsFromRange(domains_[j].max - dm, domains_[j].max, j));
281 xjBins[j].insert(xjBins[j].end(), underflowBins.begin(), underflowBins.end());
282 }
283 }
284 std::vector<int> search_box;
285 tiles.searchBox(xjBins, search_box);
286
287 // loop over all bins in the search box
288 for (int binId : search_box) {
289 // iterate over the points inside this bin
290 for (auto binIterId : tiles[binId]) {
291 int j{binIterId};
292 // query N'_{dm}(i)
293 bool foundHigher{(points_.rho[j] > rho_i)};
294 // in the rare case where rho is the same, use detid
295 foundHigher = foundHigher || ((points_.rho[j] == rho_i) && (j > i));
296 float dist_ij{distance(i, j)};
297 if (foundHigher && dist_ij <= dm) { // definition of N'_{dm}(i)
298 // find the nearest point within N'_{dm}(i)
299 if (dist_ij < delta_i) {
300 // update delta_i and nearestHigher_i
301 delta_i = dist_ij;
302 nearestHigher_i = j;
303 }
304 }
305 } // end of interate inside this bin
306 } // end of loop over bins in the search box
307
308 points_.delta[i] = delta_i;
309 points_.nearestHigher[i] = nearestHigher_i;
310 } // end of loop over points
311}

313template <uint8_t Ndim>
314void ClusteringAlgo<Ndim>::findAndAssignClusters() {
315 int nClusters{};
316
317 // find cluster seeds and outlier
318 std::vector<int> localStack; // this vector will contain the indexes of all the seeds
319 // loop over all points
320 for (int i{}; i < points_.n; ++i) {
321 // initialize clusterIndex
322 points_.clusterIndex[i] = -1;
323
324 float deltai{points_.delta[i]};
325 float rhoi{points_.rho[i]};
326
327 // determine seed or outlier
328 bool isSeed{(deltai > dc_) && (rhoi >= rhoc_)};
329 bool isOutlier{(deltai > outlierDeltaFactor_ * dc_) && (rhoi < rhoc_)};
330 if (isSeed) {
331 // set isSeed as 1
332 points_.isSeed[i] = 1;
333 // set cluster id
334 points_.clusterIndex[i] = nClusters;
335 // increment number of clusters
336 ++nClusters;
337 // add seed into local stack
338 localStack.push_back(i);
339 } else if (!isOutlier) {
340 // register as follower at its nearest higher
341 points_.followers[points_.nearestHigher[i]].push_back(i);
342 }
343 }
344
345 // expend clusters from seeds
346 while (!localStack.empty()) {
347 int i{localStack.back()};
348 auto& followers{points_.followers[i]};
349 localStack.pop_back();
350
351 // loop over followers
352 for (int j : followers) {
353 // pass id from i to a i's follower
354 points_.clusterIndex[j] = points_.clusterIndex[i];
355 // push this follower to localStack
356 localStack.push_back(j);
357 }
358 }
359}

361template <uint8_t Ndim>
362inline float ClusteringAlgo<Ndim>::distance(int i, int j) const {
363 float qSum{}; // quadratic sum
364 for (int k{}; k != Ndim; ++k) {
365 float delta_xk{
366 deltaPhi(points_.coordinates_[k][i], points_.coordinates_[k][j], domains_[k].min, domains_[k].max)};
367 qSum += std::pow(delta_xk, 2);
368 }
369
370 return std::sqrt(qSum);
371}
