88 std::array<float, Ndim> calculateTileSize(int NTiles, tiles<Ndim>& tiles_) {
89 std::array<float, Ndim> tileSizes;
90 int NperDim{static_cast<int>(std::pow(NTiles, 1.0 / Ndim))};
91
92 for (int i{}; i != Ndim; ++i) {

96 tiles_.minMax[i] = {dimMin, dimMax};
97 tileSize = (dimMax - dimMin) / NperDim;

131 void calculateLocalDensity(tiles<Ndim>& tiles, kernel const& ker) {
132 // loop over all points
133 for (int i{}; i < points_.n; ++i) {
134 // get search box
135 std::array<std::vector<int>, Ndim> xjBins;
136 for (int j{}; j != Ndim; ++j) {
137 xjBins[j] =
138 tiles.getBinsFromRange(points_.coordinates_[j][i] - dc_, points_.coordinates_[j][i] + dc_, j);
139
140 // Overflow
141 if (points_.coordinates_[j][i] + dc_ > domains_[j].max) {
142 std::vector<int> overflowBins =
143 std::move(tiles.getBinsFromRange(domains_[j].min, domains_[j].min + dc_, j));
144 xjBins[j].insert(xjBins[j].end(), overflowBins.begin(), overflowBins.end());
145 // Underflow
146 } else if (points_.coordinates_[j][i] - dc_ < domains_[j].min) {
147 std::vector<int> underflowBins =
148 std::move(tiles.getBinsFromRange(domains_[j].max - dc_, domains_[j].max, j));
149 xjBins[j].insert(xjBins[j].end(), underflowBins.begin(), underflowBins.end());
150 }
151 }
152 std::vector<int> search_box;
153 tiles.searchBox(xjBins, search_box);
154
155 // loop over bins in the search box
156 for (int binId : search_box) {
157 // iterate over the points inside this bin
158 for (auto binIterId : tiles[binId]) {
159 int j{binIterId};
160 // query N_{dc_}(i)
161 float dist_ij{distance(i, j)};
162
163 if (dist_ij <= dc_) {
164 // sum weights within N_{dc_}(i) using the chosen kernel
165 points_.rho[i] += ker(dist_ij, i, j) * points_.weight[j];
166 }
167 } // end of interate inside this bin
168 } // end of loop over bins in the search box
169 } // end of loop over points
170 }

172 void calculateDistanceToHigher(tiles<Ndim>& tiles) {
173 float dm{outlierDeltaFactor_ * dc_};
174
175 // loop over all points
176 for (int i{}; i < points_.n; ++i) {
177 // default values of delta and nearest higher for i
178 float delta_i{std::numeric_limits<float>::max()};
179 int nearestHigher_i{-1}; // if this doesn't change, the point is either a seed or an outlier
180 float rho_i{points_.rho[i]};
181
182 // get search box
183 std::array<std::vector<int>, Ndim> xjBins;
184 for (int j{}; j != Ndim; ++j) {
185 xjBins[j] =
186 tiles.getBinsFromRange(points_.coordinates_[j][i] - dm, points_.coordinates_[j][i] + dm, j);
187
188 // Overflow
189 if (points_.coordinates_[j][i] + dm > domains_[j].max) {
190 std::vector<int> overflowBins =
191 std::move(tiles.getBinsFromRange(domains_[j].min, domains_[j].min + dm, j));
192 xjBins[j].insert(xjBins[j].end(), overflowBins.begin(), overflowBins.end());
193 // Underflow
194 } else if (points_.coordinates_[j][i] - dm < domains_[j].min) {
195 std::vector<int> underflowBins =
196 std::move(tiles.getBinsFromRange(domains_[j].max - dm, domains_[j].max, j));
197 xjBins[j].insert(xjBins[j].end(), underflowBins.begin(), underflowBins.end());
198 }
199 }
200 std::vector<int> search_box;
201 tiles.searchBox(xjBins, search_box);
202
203 // loop over all bins in the search box
204 for (int binId : search_box) {
205 // iterate over the points inside this bin
206 for (auto binIterId : tiles[binId]) {
207 int j{binIterId};
208 // query N'_{dm}(i)
209 bool foundHigher{(points_.rho[j] > rho_i)};
210 // in the rare case where rho is the same, use detid
211 foundHigher = foundHigher || ((points_.rho[j] == rho_i) && (j > i));
212 float dist_ij{distance(i, j)};
213 if (foundHigher && dist_ij <= dm) { // definition of N'_{dm}(i)
214 // find the nearest point within N'_{dm}(i)
215 if (dist_ij < delta_i) {
216 // update delta_i and nearestHigher_i
217 delta_i = dist_ij;
218 nearestHigher_i = j;
219 }
220 }
221 } // end of interate inside this bin
222 } // end of loop over bins in the search box
223
224 points_.delta[i] = delta_i;
225 points_.nearestHigher[i] = nearestHigher_i;
226 } // end of loop over points
227 }

229 void findAndAssignClusters() {
230 int nClusters{};
231
232 // find cluster seeds and outlier
233 std::vector<int> localStack; // this vector will contain the indexes of all the seeds
234 // loop over all points
235 for (int i{}; i < points_.n; ++i) {
236 // initialize clusterIndex
237 points_.clusterIndex[i] = -1;
238
239 float deltai{points_.delta[i]};
240 float rhoi{points_.rho[i]};
241
242 // determine seed or outlier
243 bool isSeed{(deltai > dc_) && (rhoi >= rhoc_)};
244 bool isOutlier{(deltai > outlierDeltaFactor_ * dc_) && (rhoi < rhoc_)};
245 if (isSeed) {
246 // set isSeed as 1
247 points_.isSeed[i] = 1;
248 // set cluster id
249 points_.clusterIndex[i] = nClusters;
250 // increment number of clusters
251 ++nClusters;
252 // add seed into local stack
253 localStack.push_back(i);
254 } else if (!isOutlier) {
255 // register as follower at its nearest higher
256 points_.followers[points_.nearestHigher[i]].push_back(i);
257 }
258 }
259
260 // expend clusters from seeds
261 while (!localStack.empty()) {
262 int i{localStack.back()};
263 auto& followers{points_.followers[i]};
264 localStack.pop_back();
265
266 // loop over followers
267 for (int j : followers) {
268 // pass id from i to a i's follower
269 points_.clusterIndex[j] = points_.clusterIndex[i];
270 // push this follower to localStack
271 localStack.push_back(j);
272 }
273 }
274 }

276 inline float distance(int i, int j) const {
277 float qSum{}; // quadratic sum
278 for (int k{}; k != Ndim; ++k) {
279 float delta_xk{deltaPhi(
280 points_.coordinates_[k][i], points_.coordinates_[k][j], domains_[k].min, domains_[k].max)};
281 qSum += std::pow(delta_xk, 2);
282 }
283
284 return std::sqrt(qSum);
285 }
